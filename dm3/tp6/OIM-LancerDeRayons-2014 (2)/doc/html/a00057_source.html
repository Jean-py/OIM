488 Color reflect (Ray ray_, Isect isect_, Ray * p_ray_) {
489  Vector3D refl_dir;
490  Color fs = isect_.bsdf().reflect_dir (-ray_.dir(), &refl_dir);
491 
492  *p_ray_ = Ray (ray_, fs, isect_.dg().pos(), refl_dir);
493  // Update ray differentials on p_ray_
494 
495  Vector3D dPpdx = ray_.dPdx();
496  Vector3D dPpdy = ray_.dPdy();
497 
498 
499  float dDdotNdx = ray_.dDdx().dot(isect_.dg().normal()) + ray_.dir().dot(isect_.dg().dNdx());
500  float dDdotNdy = ray_.dDdy().dot(isect_.dg().normal()) + ray_.dir().dot(isect_.dg().dNdy());
501 
502  Vector3D dDpdx(ray_.dDdx() - ((isect_.dg().dNdx()*ray_.dir().dot(isect_.dg().normal()) + isect_.dg().normal() * dDdotNdx) * 2.f) );
503  Vector3D dDpdy(ray_.dDdy() - ((isect_.dg().dNdy()*ray_.dir().dot(isect_.dg().normal()) + isect_.dg().normal() * dDdotNdy) * 2.f) );
504 
505  p_ray_->setdPdx(dPpdx);
506  p_ray_->setdPdy(dPpdy);
507  p_ray_->setdDdx(dDpdx);
508  p_ray_->setdDdy(dDpdy);
509 
510  return fs * fabsf (isect_.dg().normal().dot (refl_dir)); //cos(theta) here for easier use
511 }

520 Color refract (Ray ray_, Isect isect_, Ray * p_ray_) {
521 
522  Vector3D refr_dir;
523  Color fs = isect_.bsdf().refract_dir (-ray_.dir(), &refr_dir);
524 
525  *p_ray_ = Ray (ray_, fs, isect_.dg().pos(), refr_dir);
526  //Update ray differentials on p_ray_
527 
528  bool entering = (ray_.dir().dot(isect_.dg().normal()) > 0.f);
529  float ior=isect_.bsdf().getIor(entering);
530  float mu = ior * ray_.dir().dot(isect_.dg().normal()) - refr_dir.dot(isect_.dg().normal());
531 
532  Vector3D dPpdx = ray_.dPdx();
533  Vector3D dPpdy = ray_.dPdy();
534  float dDdotNdx = ray_.dDdx().dot(isect_.dg().normal()) + ray_.dir().dot(isect_.dg().dNdx());
535  float dDdotNdy = ray_.dDdy().dot(isect_.dg().normal()) + ray_.dir().dot(isect_.dg().dNdy());
536 
537  float dmudx = (ior - (ior*ior*ray_.dir().dot(isect_.dg().normal())/ refr_dir.dot(isect_.dg().normal())))*dDdotNdx;
538  float dmudy = (ior - (ior*ior*ray_.dir().dot(isect_.dg().normal())/ refr_dir.dot(isect_.dg().normal())))*dDdotNdy;
539 
540 
541  Vector3D dDpdx( ray_.dDdx() * ior - (isect_.dg().dNdx() * mu + isect_.dg().normal() * dmudx) );
542  Vector3D dDpdy( ray_.dDdy() * ior - (isect_.dg().dNdy() * mu + isect_.dg().normal() * dmudy) );
543 
544  p_ray_->setdPdx(dPpdx);
545  p_ray_->setdPdy(dPpdy);
546  p_ray_->setdDdx(dDpdx);
547  p_ray_->setdDdy(dDpdy);
548 
549  return fs * fabsf (isect_.dg().normal().dot (refr_dir)); //cos(theta) here for easier use
550 }
