98  }

101  case BLINN_PHONG:
102  {
103  if (m_normal.dot (to_eye_) * m_normal.dot (to_light_) < 0.f) {
104  //not same hemisphere
105  return Color (0.f);
106  }
107  //www.cs.ucl.ac.uk/staff/j.kautz/GameCourse/04_PointLights.pdf
108  Vector3D half = (to_eye_ + to_light_).normalize();
109  return m_kd + m_ks * (std::pow ((half.dot (m_normal)), m_shininess) * (m_shininess * 4) / (8.f * M_PI));
110  break;
111  }
112  case GLOSSY:
113  {
114  if (m_normal.dot (to_eye_) * m_normal.dot (to_light_) < 0.f) {
115  //not same hemisphere
116  return Color (0.f);
117  }
118  float r0 = (1-m_ior)/(1+m_ior);
119  r0 = r0 * r0;
120  Vector3D half = (to_eye_ + to_light_).normalize();
121  float costetha = half.dot(to_eye_);
122  float r = r0 + (1-r0)*std::pow((1-costetha), 5.);
123  //www.cs.ucl.ac.uk/staff/j.kautz/GameCourse/04_PointLights.pdf
124  return (m_kd*(1./(M_PI)) + m_ks * r * (std::pow ((half.dot (m_normal)), m_shininess) * (m_shininess * 4) / (8.f * M_PI)));
