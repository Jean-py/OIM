83  Vector3D computenormalDifferential(const Vector3D & p_, Vector3D &dPpdx, Vector3D &dPpdy, Vector3D &dNdx, Vector3D &dNdy) const{

90

98  dNdy = dnpdy*std::pow(ndotn, -0.5f) + n * fnpy;

103  int get_clip_points(Ray& ray_, Diff_Geom ipoints[2], float t[2]) const{
104  int numintersection = 0;
105  Vector3D E = ray_.ori() - m_vertex;
106  float AdD = m_axis.dot(ray_.dir());
107  float cosSqr = m_costheta*m_costheta;
108  float AdE = m_axis.dot(E);
109  float DdE = ray_.dir().dot(E);
110  float EdE = E.dot(E);
111  float c2 = AdD*AdD-cosSqr;
112  float c1 = AdD*AdE - cosSqr*DdE;
113  float c0 = AdE*AdE - cosSqr*EdE;
114  float dot;
115  Vector3D zero;
116 
117  if (std::fabs(c2)>0) {
118  float discr = c1*c1 - c0*c2;
119  float invC2 = 1.f/c2;
120  if (discr < 0) {
121  // No intersection
122  return 0;
123  } else if (discr > 0) {
124  // two distinct intersections
125  float root = std::sqrt(discr);
126  // entry point
127  t[numintersection] = (-c1 + root) * invC2;
128  E = ray_.at(t[numintersection]) - m_vertex;
129  dot = E.dot(m_axis);
130  if ( (dot > 0) && (dot <= m_height))
131  {
132  fillConicalDiffGeom(ipoints[numintersection], ray_, t[numintersection], true);
133  ++numintersection;
134  }
135  // exit point
136  t[numintersection] = (-c1 - root) * invC2;
137  E = ray_.at(t[numintersection]) - m_vertex;
138  dot = E.dot(m_axis);
139  if ( (dot > 0) && (dot <= m_height))
140  {
141  fillConicalDiffGeom(ipoints[numintersection], ray_, t[numintersection], false);
142  ++numintersection;
143  }
144  return numintersection;
145  } else {
146  // one reapeated intersection : ray is tangent to the cone
147  // may be return 0 instead of an intersection ?
148  return 0;
149  t[numintersection] = -c1 * invC2;
150  E = ray_.at(t[numintersection]) - m_vertex;

236  bool intersect (Ray& ray_, Diff_Geom * p_diff_geom_) const {

238  float t[2];
239 
240  int numintersection = get_clip_points(ray_, ipoints, t);
241  /* numintersection == 2 -> two intersections with the cone above vertex :
242  * return the smallest t
243  * numintersection == 1 -> one intersection with the cone above vertex :
244  * check cap plane (that will add an intersection point) before choosing the smalest positive t
245  * numintersection == 0 -> no intersection
246  */
247  if (numintersection == 0)
248  return false;
249  else if (numintersection == 2) {
250  int i=0;
251  while ( (i<numintersection) && (t[i] < ray_.ray_epsilon()) )
252  i++;
253  if ( (i < numintersection) && (t[i] < ray_.max_t()) ) {
254  *p_diff_geom_ = ipoints[i];
255  ray_.set_max_t (t[i]);
256  return true;
257  } else
258  return false;
259  } else {
260  // 1 intersection
261  // check with cap plane
262  Plane cap(m_vertex + m_axis * m_height, m_axis);
263  Diff_Geom plane_intersection;
264  Ray r(ray_);
265  if (cap.intersect(ray_, &plane_intersection)) {
266  if (ray_.max_t() < t[0]) {
267  // beware of in/out status
268  *p_diff_geom_ = plane_intersection;
269  float u, v;

271  v = p_diff_geom_->v()/(PSCALE*m_radius);
272  p_diff_geom_->set_u(u);
273  p_diff_geom_->set_v(v);
274  Vector3D dTdx = p_diff_geom_->dTdx() * (1.f/(PSCALE*m_radius));

276  p_diff_geom_->set_dTdx(dTdx);
277  p_diff_geom_->set_dTdy(dTdy);
278  return true;
279  }

282  // beware of in/out status
283  *p_diff_geom_ = ipoints[0];
284  ray_.set_max_t(t[0]);
