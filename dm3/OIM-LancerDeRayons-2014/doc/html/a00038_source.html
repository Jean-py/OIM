48  void computeCylindricalTexInfo(const Vector3D &pt_inter, const Vector3D &normal, const Vector3D &dPpdx, const Vector3D &dPpdy, float &u, float &v, Vector3D &dTdx, Vector3D &dTdy) const{
49  Vector3D diff = pt_inter - m_base;
50  Vector3D ptOnCylinder(U.dot(diff), V.dot(diff), W.dot(diff));
51  ptOnCylinder[0] = ptOnCylinder[0]/m_radius;
52  ptOnCylinder[1] = ptOnCylinder[1]/m_radius;
53  // texture coordinates
54  u = (std::atan2(ptOnCylinder[1],ptOnCylinder[0])+M_PI)/(2.f*M_PI);
55  v = ((ptOnCylinder[2]+m_length)*0.5/m_length);
56  // compute texture differential
57  dTdx[0] = (1.f/(2.f*M_PI)) * (1.f/(1.f + std::pow(ptOnCylinder[1]/ptOnCylinder[0], 2.f)))*( (V.dot(dPpdx)*ptOnCylinder[0] - U.dot(dPpdx)*ptOnCylinder[1]) / std::pow(ptOnCylinder[0], 2.f) );
58  dTdy[0] = (1.f/(2.f*M_PI)) * (1.f/(1.f + std::pow(ptOnCylinder[1]/ptOnCylinder[0], 2.f)))*( (V.dot(dPpdy)*ptOnCylinder[0] - U.dot(dPpdy)*ptOnCylinder[1]) /std::pow(ptOnCylinder[0], 2.f) );
59  float coef = (1.f / (2.f*m_length));
60  dTdx[1] = coef * W.dot(dPpdx);
61  dTdy[1] = coef * W.dot(dPpdy);
62  }

65  Vector3D pt_inter = r.at(t);
66  Vector3D dPpdx, dPpdy;
67 
68  Vector3D cp = pt_inter - m_base;
69  Vector3D ap = m_axis.cross_product(cp);
70  Vector3D np = ap.cross_product(m_axis);
71  float npdotnp = np.dot(np);
72  Vector3D normal = np.normalize();
73  r.propagateDifferentials(t, normal, dPpdx, dPpdy);
74 
75  Vector3D dnpdx = (m_axis.cross_product(dPpdx)).cross_product(m_axis);
76  Vector3D dnpdy = (m_axis.cross_product(dPpdy)).cross_product(m_axis);
77  float fnpx = -std::pow(npdotnp, -1.5f) * dnpdx.dot(np);
78  float fnpy = -std::pow(npdotnp, -1.5f) * dnpdy.dot(np);
79  // compute Ndx and dNdy
80  Vector3D dNdx, dNdy;
81  dNdx = dnpdx*std::pow(npdotnp, -0.5f) + np * fnpx;
82  dNdy = dnpdy*std::pow(npdotnp, -0.5f) + np * fnpy;
83  // compute texinfo
84  Vector3D dTdx;
85  Vector3D dTdy;

92  int get_clip_points(Ray& ray_, Diff_Geom ipoints[2], float t[2]) const{
93  Vector3D zero;
94  float rSqr = m_radius*m_radius;
95 
96  // convert incoming line origin to cylinder coordinates
97  Vector3D diff = ray_.ori() - m_base;
98  Vector3D P(U.dot(diff), V.dot(diff), W.dot(diff));
99  // Get the z-value, in cylinder coordinates, of the incoming line's
100  // unit-length direction.
101  float dz = W.dot(ray_.dir());
102 
103  if (std::fabs(dz) >= 1.f - ray_.ray_epsilon()) {
104  // The line is parallel to the cylinder axis. Determine if the line
105  // intersects the cylinder end disks.
106  float radialSqrDist = rSqr - P.x()*P.x() - P.y()*P.y();
107  if (radialSqrDist < 0.f) { // epsilon ?
108  // Line outside the cylinder, no intersection.
109  return 0; // 0
110  }
111  // Line intersects the cylinder end disks.
112  // todo verify normals
113  if (dz > 0.f) {
114  t[0] = -P.z() - m_length;
115  fillPlanarDiffGeom(ipoints[0], ray_, t[0], m_axis, true);
116  t[1] = -P.z() + m_length;
117  fillPlanarDiffGeom(ipoints[1], ray_, t[1], -m_axis, false);
118 
119  } else {
120  t[0] = P.z() - m_length;
121  fillPlanarDiffGeom(ipoints[0], ray_, t[0], -m_axis, true);
122  t[1] = P.z() + m_length;
123  fillPlanarDiffGeom(ipoints[1], ray_, t[1], m_axis, false);
124  }
125  return 2;
126  }
127 
128  // convert incoming line unit-length direction to cylinder coordinates
129  Vector3D D(U.dot(ray_.dir()),V.dot(ray_.dir()),dz);
130  float a0, a1, a2, discr, root, inv, tValue;
131  if (std::fabs(D.z()) <= ray_.ray_epsilon()) {
132  // The line is perpendicular to the cylinder axis.
133  if (std::fabs(P.z()) > m_length) {
134  // Line is outside the planes of the cylinder end disks.
135  return 0;
136  }
137  // Test intersection of line P+t*D with infinite cylinder
138  // x^2+y^2 = r^2. This reduces to computing the roots of a
139  // quadratic equation. If P = (px,py,pz) and D = (dx,dy,dz),
140  // then the quadratic equation is
141  // (dx^2+dy^2)*t^2 + 2*(px*dx+py*dy)*t + (px^2+py^2-r^2) = 0
142  a0 = P.x()*P.x() + P.y()*P.y() - rSqr;
143  a1 = P.x()*D.x() + P.y()*D.y();
144  a2 = D.x()*D.x() + D.y()*D.y();
145  discr = a1*a1 - a0*a2;
146  if (discr < 0.f) {
147  // Line does not intersect cylinder.
148  return 0;
149  }
150  else if (discr > 0) { // epsilon ?
151  // Line intersects cylinder in two places.
152  root = std::sqrt(discr);
153  inv = 1.f/a2;
154  t[0] = (-a1 - root)*inv;
155  fillCylindricalDiffGeom(ipoints[0], ray_, t[0], true);
156  t[1] = (-a1 + root)*inv;
157  fillCylindricalDiffGeom(ipoints[1], ray_, t[1], false);
158  return 2;
159  }
160  else
161  {
162  // Line is tangent to the cylinder.
163  //t[0] = -a1/a2;
164  //return 1;
165  return 0;
166  }
167  }
168  // Test plane intersections first.
169  int quantity = 0;
170  inv = 1.0f/D.z();
171 
172  float t0 = (-m_length - P.z())*inv;
173  float xTmp = P.x() + t0*D.x();
174  float yTmp = P.y() + t0*D.y();
175  if (xTmp*xTmp + yTmp*yTmp <= rSqr) {
176  // Planar intersection inside the top cylinder end disk.
177  t[quantity] = t0;
178  fillPlanarDiffGeom(ipoints[quantity], ray_, t[quantity], m_axis, true);
179  ++quantity;
180  }
181 
182  float t1 = (+m_length - P.z())*inv;
183  xTmp = P.x() + t1*D.x();
184  yTmp = P.y() + t1*D.y();
185  if (xTmp*xTmp + yTmp*yTmp <= rSqr) {
186  // Planar intersection inside the bottom cylinder end disk.
187  t[quantity] = t1;
188  fillPlanarDiffGeom(ipoints[quantity], ray_, t[quantity], -m_axis, false);
189  ++quantity;
190  }
191 
192  if (quantity == 2) {
193  // Line intersects both top and bottom cylinder end disks.
194  if (t[0] > t[1]) {
195  float save = t[0];
196  t[0] = t[1];
197  t[1] = save;
198  Diff_Geom gds = ipoints[0];
199  ipoints[0] = ipoints[1];
200  ipoints[1] = gds;
201  ipoints[0].set_in(true);
202  ipoints[1].set_in(false);
203  }
204  return 2;
205  }
206  // If quantity == 1, then the line must intersect cylinder wall in a
207  // single point somewhere between the end disks. This case is detected
208  // in the following code that tests for intersection between line and
209  // cylinder wall.
210  a0 = P.x()*P.x() + P.y()*P.y() - rSqr;
211  a1 = P.x()*D.x() + P.y()*D.y();
212  a2 = D.x()*D.x() + D.y()*D.y();
213  discr = a1*a1 - a0*a2;
214  if (discr < 0.f) {
215  // Line does not intersect cylinder wall.
216  //assertion(quantity == 0, "Unexpected condition\n");
217  return 0;
218  } else if (discr > 0.f) {
219  root = std::sqrt(discr);
220  inv = 1.f/a2;
221  tValue = (-a1 - root)*inv;
222  if (t0 <= t1) {
223  if (t0 <= tValue && tValue <= t1) {
224  t[quantity] = tValue;
225  fillCylindricalDiffGeom(ipoints[quantity], ray_, t[quantity], false);
226  ++quantity;
227  }
228  } else {
229  if (t1 <= tValue && tValue <= t0) {
230  t[quantity] = tValue;
231  fillCylindricalDiffGeom(ipoints[quantity], ray_, t[quantity], false);
232  ++quantity;
233  }
234  }
235  if (quantity == 2) {
236  // Line intersects one of the cylinder end disks and once on the
237  // cylinder wall.
238  if (t[0] > t[1]) {
239  float save = t[0];
240  t[0] = t[1];
241  t[1] = save;
242  Diff_Geom gds = ipoints[0];
243  ipoints[0] = ipoints[1];
244  ipoints[1] = gds;
245  ipoints[0].set_in(true);
246  ipoints[1].set_in(false);
247  }
248 
249  return 2;
250  }
251  tValue = (-a1 + root)*inv;
252  if (t0 <= t1) {
253  if (t0 <= tValue && tValue <= t1) {
254  t[quantity] = tValue;
255  fillCylindricalDiffGeom(ipoints[quantity], ray_, t[quantity], false);
256  ++quantity;
257  }
258  } else {
259  if (t1 <= tValue && tValue <= t0) {
260  t[quantity] = tValue;
261  fillCylindricalDiffGeom(ipoints[quantity], ray_, t[quantity], false);
262  ++quantity;
263  }
264  }
265  } else {
266  tValue = -a1/a2;
267  if (t0 <= t1) {
268  if (t0 <= tValue && tValue <= t1) {
269  t[quantity] = tValue;
270  fillCylindricalDiffGeom(ipoints[quantity], ray_, t[quantity], false);
271  ++quantity;
272  }
273  } else {
274  if (t1 <= tValue && tValue <= t0) {
275  t[quantity] = tValue;
276  fillCylindricalDiffGeom(ipoints[quantity], ray_, t[quantity], false);
277  ++quantity;
278  }
279  }
280  }
281  if (quantity == 2) {
282  if (t[0] > t[1]) {
283  float save = t[0];
284  t[0] = t[1];
285  t[1] = save;
