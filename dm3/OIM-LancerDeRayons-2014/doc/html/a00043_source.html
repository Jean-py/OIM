66  void write_to_exr_file (const string& file_name_) {
67  Header header (m_x_res, m_y_res);
68 
69  //edit the active zone.
70  Box2i data_window (V2i (0, 0),
71  V2i (m_x_res - 1, m_y_res - 1));
72 
73  header.dataWindow() = data_window; //beuark.
74 
75  header.channels().insert ("R", Channel (HALF));
76  header.channels().insert ("G", Channel (HALF));
77  header.channels().insert ("B", Channel (HALF));
78 
79  const int x_count = m_x_res;
80  const int nb_pixels = m_x_res * m_y_res;
81 
82  half * half_rgb = new half[3 * nb_pixels];
83 
84  int offset = 0;
85  int num_pixel = 0;
86 
87  for (int y = 0; y < m_y_res; y++) {
88  for (int x = 0; x < m_x_res; x++, num_pixel++) {
89  Color color = pixel (x, m_y_res-1-y);
90 
91  for (int i = 0; i < 3; i++, offset++) {
92  half_rgb[offset] = color[i];
93  }
94  }
95  }
96 
97  offset = 0;
98 
99  half_rgb -= 3 * offset;
100 
101  FrameBuffer fb;
102  //there are 3 * sizeof(half) bytes between two R elements.
103  fb.insert ("R", Slice (HALF, (char *)half_rgb, 3 * sizeof (half),
104  3 * x_count * sizeof (half)));
105  //the first element of G is sizeof(half) after the first element of R.
106  fb.insert ("G", Slice (HALF, (char *)half_rgb + sizeof(half), 3 * sizeof (half),
107  3 * x_count * sizeof (half)));
108  //the first B element is 2 * sizeof (half) bytes after the first element of R.
109  fb.insert ("B", Slice (HALF, (char *)half_rgb + 2 * sizeof(half), 3 * sizeof (half),
110  3 * x_count * sizeof (half)));
111  try {
112  OutputFile file (file_name_.c_str(), header);
113  file.setFrameBuffer (fb);
114  //y_count() rows to write
115  file.writePixels (m_y_res);
116  } catch (const std::exception &e) {
117  std::cerr<<"Unable to write image file "<<file_name_<<" : "<<e.what()<<std::endl;
118  }
119 
120  //release the memory, but come back to the real address before.
121  delete[] (half_rgb + 3 * offset);
122  }

166  void postProcess(const Color *pixels, unsigned char *rgb_chanels, int width, int height){
167  double *xyY_chanels = new double [width*height*3];
168  double keyValue = KEYVALUE;
169  double delta = 1e-6f;
170  double avLum = 0.0f;
171  /* sRGB D65 matrix from http://www.brucelindbloom.com/index.html?Eqn_RGB_XYZ_Matrix.html */
172  double rgbToXyzMatrix[3][3] = {
173  {0.4124564, 0.3575761, 0.1804375},
174  {0.2126729, 0.7151522, 0.0721750},
175  {0.0193339, 0.1191920, 0.9503041}
176 
177  };
178  /* sRGB D65 matrix from http://www.brucelindbloom.com/index.html?Eqn_RGB_XYZ_Matrix.html */
179  double xyzToRgbMatrix[3][3] = {
180  {3.2404542, -1.5371385, -0.4985314},
181  {-0.9692660, 1.8760108, 0.0415560},
182  {0.0556434, -0.2040259, 1.0572252}
183  };
184 
185 
186 #define X xyz[0]
187 #define Y xyz[1]
188 #define Z xyz[2]
189  int maxI = width*height;
190  for (int i=0; i<maxI; ++i) {
191  double w;
192  // Transform from RGB to XYZ
193  Color pixel = pixels[i];
194  double xyz[3] = {0.0, 0.0, 0.0};
195  for (int ii = 0; ii < 3; ++ii)
196  for (int jj = 0; jj < 3; ++jj)
197  xyz[ii] += rgbToXyzMatrix[ii][jj] * (double)(pixel[jj]);
198 
199 
200  // Transform from XYZ to xyY
201  w = (X + Y + Z);
202  if(w > 0.) {
203  xyY_chanels[i*3 ] = X/w; // x
204  xyY_chanels[i*3 + 1] = Y/w; // y
205  xyY_chanels[i*3 + 2] = Y; // Y
206  avLum += std::log(std::max(Y, 0.) + delta);
207  } else {
208  xyY_chanels[i*3 ] = 0.; // x
209  xyY_chanels[i*3 + 1] = 0.; // y
210  xyY_chanels[i*3 + 2] = 0.; // Y
211  }
212  }
213  avLum = exp(avLum/(double)(width*height));
214 
215  maxI*=3;
216  for(int i=0; i<maxI; i+=3) {
217  double xyz[3] = {0.0, 0.0, 0.0};
218  // tonemap
219  double lum = std::max(xyY_chanels[i+2], 0.);
220  lum = (keyValue*lum)/avLum;
221  float lumWhite = avLum*100000.;
222  xyY_chanels[i+2] = lum*(1.0+(lum/(lumWhite*lumWhite)))/(1.0+lum);//lum/(1.0+lum);
223 
224  // Transform from xyY to XYZ
225  Y = xyY_chanels[i+2] ;
226 
227  //if((Y > 0.0) && (xyY_chanels[i] > 0.0) && (xyY_chanels[i+1] > 0.0)) {
228  if((xyY_chanels[i] > 0.0) && (xyY_chanels[i+1] > 0.0)) {
229  X = (Y * xyY_chanels[i]) / xyY_chanels[i+1];
230  Z = (X / xyY_chanels[i]) - X - Y;
231  }
232  else {
233  X = Z = 0.0;
234  }
235 
236  // conversion XYZ vers RGB
237  double rgb[3] = {0.0, 0.0, 0.0};
238  for (int ii = 0; ii < 3; ++ii){
239  for (int jj = 0; jj < 3; ++jj)
240  rgb[ii] += xyzToRgbMatrix[ii][jj] * xyz[jj];
241  rgb[ii] = std::max(rgb[ii],0.0);
242  }
243 
244 
245  // gamma correction
246  float gamma_corr = 1.f/2.4f;
247  for (int k=0;k<3;k++)
248  if (rgb[k] <= 0.0031308f)
249  rgb[k] = 12.92f*rgb[k];
250  else
251  rgb[k] = (1.f + 0.055f) * std::pow(rgb[k],gamma_corr) - 0.055f;
252 
253  rgb_chanels[i ] = (unsigned char)(std::min(rgb[0]*255.,255.));
254  rgb_chanels[i + 1] = (unsigned char)(std::min(rgb[1]*255.,255.));
255  rgb_chanels[i + 2] = (unsigned char)(std::min(rgb[2]*255.,255.));
256  }
257 #undef X
258 #undef Y
259 #undef Z
260  delete[] xyY_chanels;
261  }
